## window 下批量重名脚本

```shell script
@echo off
setlocal EnableDelayedExpansion

set a=00
set /p name=请输入文件名:

for /f  "delims=" %%n in ('dir /a-d /b /oen .\*.*') do (
 if not "%%~nn"=="%~n0" (
  set /A a+=1
  if defined name (
   ren "%%n" "%name%-!a!%%~xn"
  ) else (
   ren "%%n" "!a!%%~xn"
  )
 )
)
exit
```
